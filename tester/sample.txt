include<main>

int factorial(int n) {
    if (n <= 1) {
        return 1;
    }
    return n * factorial(n - 1);
}

main {
    int result = factorial(5);
    print("Factorial:", result);
    
    for (int i = 0; i < 3; i=i+1) {
        if (i == 1 && i == 3) {
            break;
        }
        print(i);
    }
}
