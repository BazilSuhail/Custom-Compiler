include<main>

void displayMessage(string message);

void displayMessage(string message){
    print(message);
}

char processCharacter(char inputChar){
    switch (inputChar) {
        case 'c'{
            print("Character is c");
            break;            
        }
        case 'x'{
            print("Character is x"); 
            break;
        }
        default{
            print("Unknown character");
        }
    }
    return inputChar;
}

int calculateSum(int firstNumber, int secondNumber) {
    int sumResult = firstNumber + secondNumber;
    return sumResult;
}

int calculateProduct(int firstNumber, int secondNumber) {
    int productResult = firstNumber * secondNumber;
    return productResult;
}

void demonstrateArithmetic(){
    int sumResult = calculateSum(25, 5);
    print("25 + 5 = ");
    print(sumResult);
    
    int productResult = calculateProduct(105, 10);
    print("105 * 10 = ");
    print(productResult);
    
    float decimalResult = 105.2 + 0.33;
    print("105.2 + 0.33 = ", decimalResult);
}

void demonstrateBitwise(){
    int number1 = 5;
    int number2 = 3;

    int bitwiseAnd = number1 & number2;
    int bitwiseOr = number1 | number2;
    int bitwiseXor = number1 ^ number2;
    int leftShift = number1 << 2;
    int rightShift = number1 >> 1;

    print("Bitwise AND: 5 & 3 = ");
    print(bitwiseAnd);
    print("Bitwise OR: 5 | 3 = ");
    print(bitwiseOr);
    print("Bitwise XOR: 5 ^ 3 = ");
    print(bitwiseXor);
    print("Left shift: 5 << 2 = ");
    print(leftShift);
    print("Right shift: 5 >> 1 = ");
    print(rightShift);
}
 
main {
    displayMessage("=== Starting Program Execution ===");
    displayMessage("Welcome to the Toy Language Demo!");
    
    char processedChar = processCharacter('1');
    print("Processed character: ");
    print(processedChar);
    
    demonstrateArithmetic();
    demonstrateBitwise();

    print("Do-while loop example:");
    bool continueLoop = true;
    int counter = 1;
    
    do {
        if (counter > 4) {
            continueLoop = false;
            break;
        }
        print("Counter value: ");
        print(counter/2);
        counter = counter + 1;
    } while (true);
    
    print("\nWhile loop example:\n");
    int number = 0;
    while (number < 5){
        print("Current number: ");
        print(number);
        number = number + 1;
    }

    print("For loop example:");
    for (int index = 0; index <= 3; index++) {
        if (index == 1 || index == 3) {
            print("Special value found at index: ");
            print(index);
        }
        print("Processing index: ");
        print(index);
    }
    
    displayMessage("=== Program Execution Complete ===");
}
