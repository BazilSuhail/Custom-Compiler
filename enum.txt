// ===== 1. ENUM DECLARATIONS =====
// Basic enum declaration (global scope)
enum Color { Red, Green, Blue };

// Enum with multiple values
enum Status { Pending, Active, Completed, Failed };

// Enum for days of week
enum Day { Monday, Tuesday, Wednesday, Thursday, Friday, Saturday, Sunday };

// Simple enum
enum Direction { North, South, East, West };

// Enum for boolean-like states
enum State { On, Off };


// ===== 2. FUNCTION PROTOTYPES WITH ENUM TYPES =====
// Function returning enum type

// Function taking enum parameter
void printColor(Color c);


// Function with multiple enum parameters
int compareColors(Color c1, Color c2);


// ===== 3. FUNCTION DEFINITIONS WITH ENUM TYPES =====

// Function with enum parameter
void printColor(Color c) {
    print("Color received");
}
 

// Function with multiple parameters including enums
int compareColors(Color c1, Color c2) {
    return 1;
}

// ===== 4. MAIN FUNCTION WITH ENUM USAGE =====
main {
    // --- Variable declarations with enum types ---
    Color myColor = Red;
    Color favoriteColor = Blue;
    Status taskStatus = Pending;
    Direction heading = North;
    Day today = Monday;
    State power = On;
    
    // --- Using enum variables in expressions ---
    Color resultColor = myColor;
    
    // --- Enum variables in function calls ---
    printColor(myColor);
    printColor(Blue);
    

    // --- Enum in if statements ---
    if (taskStatus == Pending) {
        print("Task is pending");
        taskStatus = Active;
    }
    
    if (myColor == Red) {
        print("Color is Red!");
    }
    
    // --- Enum in if-else ---
    if (power == On) {
        print("Power is ON");
    } else {
        print("Power is OFF");
    }
    
    // --- Nested if with enums ---
    if (today == Monday) {
        if (taskStatus == Active) {
            print("Working on Monday");
        }
    }
    
    // --- While loop with enum ---
    while (taskStatus == Active) {
        print("Processing...");
        taskStatus = Completed;
    }
    
    // --- For loop with enum variable ---
    for (int i = 0; i < 5; i++) {
        Direction dir = North;
        print("Loop iteration");
    }
    
    // --- Switch statement with enum ---
    switch (myColor) {
        case Red:
            print("Red color selected");
            break;
        case Green:
            print("Green color selected");
            break;
        case Blue:
            print("Blue color selected");
            break;
        default:
            print("Unknown color");
    }
    
    // --- Switch with Status enum ---
    switch (taskStatus) {
        case Pending:
            print("Task pending");
            taskStatus = Active;
            break;
        case Active:
            print("Task active");
            taskStatus = Completed;
            break;
        case Completed:
            print("Task completed");
            break;
        case Failed:
            print("Task failed");
            break;
    }
    
    // --- Multiple enum variables ---
    Color color1 = Red;
    Color color2 = Green;
    Color color3 = Blue;
    
    // --- Reassigning enum variables ---
    myColor = Green;
    myColor = Blue;
    taskStatus = Failed;
    
    // --- Enum in do-while ---
    do {
        print("Checking status");
        taskStatus = Completed;
    } while (taskStatus == Active);
    
    // --- Complex expressions with enums ---
    if (myColor == Red && taskStatus == Completed) {
        print("Red and Completed");
    }
    
    if (heading == North || heading == South) {
        print("Vertical direction");
    }
    
    // --- Enum variables in nested blocks ---
    {
        Color blockColor = Red;
        if (blockColor == Red) {
            print("Block scope enum");
        }
    }
    
    // --- Print with enum context ---
    print("Final color:", myColor);
    print("Final status:", taskStatus);
    print("Direction:", heading);
}


// ===== 5. ADDITIONAL FUNCTION EXAMPLES =====
// Function with enum in for loop
void processAllColors() {
    for (int i = 0; i < 3; i++) {
        Color c = Red;
        printColor(c);
    }
}

// Function with enum in while loop
void waitForCompletion(Status s) {
    while (s == Active) {
        print("Still active");
        s = Completed;
    }
}

// Function with enum in switch
void handleDirection(Direction d) {
    switch (d) {
        case North:
            print("Going North");
            break;
        case South:
            print("Going South");
            break;
        case East:
            print("Going East");
            break;
        case West:
            print("Going West");
            break;
    }
}


// ===== SUMMARY OF ENUM SYNTAX =====
// 1. Declaration: enum Name { Value1, Value2, Value3 };
// 2. Variable: EnumType varName = EnumValue;
// 3. Function parameter: void func(EnumType param);
// 4. Function return: EnumType func() { return EnumValue; }
// 5. Assignment: enumVar = EnumValue;
// 6. Comparison: if (enumVar == EnumValue)
// 7. Switch: switch(enumVar) { case EnumValue: ... }
// 8. All enum values are identifiers (no numeric values)
