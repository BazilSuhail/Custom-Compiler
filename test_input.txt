enum State { IDLE, RUNNING, STOPPED };
const int a = 10;
int square(int n) {
    return n * n;
}

main {
    int x = 10;
    float pi = 3.14;
    bool active = true;
    State s = RUNNING;

    if (active && x > 5) {
        print("Active", x);
    }

    int i = 0;
    while (i < 3) {
        print(i);
        i++;
    }

    do {
        x--;
    } while (x > 8);

    for (int k = 0; k < 2; k++) {
        print(square(k));
    }

    switch (s) {
        case IDLE:    print("Idle"); break;
        case RUNNING: print("Running"); break;
        default:      print("Unknown");
    }
}
