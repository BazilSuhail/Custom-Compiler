// Test file for TYPE ERRORS
// This file contains instances of all 17 type error types

enum Color { Red, Green };

// 1. ErroneousVarDecl: Variable declared as void
void invalidVar;

int add(int a, int b) {
    return a + b;
}

void voidFunc() {
    print("hello");
}

main {
    int i = 10;
    float f = 3.14;
    bool b = true;

    // 2. FnCallParamCount: Calling 'add' (2 params) with 1 arg
    add(10);

    // 3. FnCallParamType: Passing float to int parameter
    add(10, f);

    // 4. ErroneousReturnType (Checked inside functions, but let's demo more)
    // 5. ExpressionTypeMismatch: Assigning float to int
    i = 3.14;

    // 6. ErroneousBreak: break outside loop or switch
    break;

    // 7. NonBooleanCondStmt: if condition is int, not bool
    if (i) {
        print("wont work");
    }

    // 8. AttemptedBoolOpOnNonBools: && on integers
    int result = 10 && 20;

    // 9. AttemptedBitOpOnNonInt: & on floats
    float bitResult = f & 2.2;

    // 10. AttemptedShiftOnNonInt: << on floats
    float shiftResult = f << 2;

    // 11. AttemptedAddOpOnNonNumeric: Arithmetic on non-numeric (if supported in grammar)
    // Note: Assuming 'print' is a statement, not an expression.

    // 12. ReturnStmtInVoid: Handled in a separate function below
    
    // 13. NonBooleanSwitchExpression: switch on float
    switch (f) {
        case 1.0: break;
    }

    // 14. InvalidCaseValueType: switch int, case char (strict check)
    switch (i) {
        case 'a': break; 
    }

    // 15. IncrementDecrementOnNonInt: float increment
    f++;

    // 16. NotOnNonBool: ! on int
    bool notResult = !i;

    // 17. AttemptedAddOpOnNonNumeric: Demo with mismatched numeric types (strict)
    int addMismatch = i + f;
}
