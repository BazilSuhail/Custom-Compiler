main {
    // --- PART 1: OPTIMIZATION CHECK ---
    // The optimizer should resolve all of these at compile-time.
    
    // 1. Constant Folding: 100 / 2 + 50 * 2 -> 50 + 100 -> 150
    int fold = 100 / 2 + 50 * 2; 

    // 2. Constant Propagation: x is 10, so y = 10 + 5 -> 15
    const int x = 10;
    int y = x + 5; 

    // 3. Copy Propagation: c2 -> c1 -> fold. 
    // result = 150 + 15 -> 165
    int c1 = fold;
    int c2 = c1;
    int result = c2 + y;

    // 4. Peephole (Algebraic Identities)
    int base = 77;
    int id1 = base + 0;   // base
    int id2 = id1 * 1;    // base
    int id3 = id2 - id2;  // 0
    int id4 = base / 1;   // base

    // 5. Dead Code Elimination
    int unused_var = 999;    // This should be pruned from the optimized TAC
    int partial = id1 + id3; // Should simplify to: partial = base + 0 -> base

    print("--- Optimization Results ---");
    print("Folded (150):", fold);
    print("Propagated (15):", y);
    print("Final Result (165):", result);
    print("Identities (77, 0, 77):", id1, id3, id4);
    print("");

    // --- PART 2: WHILE LOOP (Right Triangle) ---
    print("--- Printing Right Triangle ---");
    
    for (int i = 1; i <= 5; ++i) {
        // Inner loop to print stars in each row
        for (int j = 1; j <= i; ++j) {
            print("* ");
        }
        // Move to the next line after printing each row
        print("\n");
    }

    print("------------------------------");
    print("Test Execution Complete.");
}
