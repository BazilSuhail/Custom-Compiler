// Sample program demonstrating the updated grammar

// Global variable declarations
int globalCounter = 0;
float pi = 3.14159;

// Function declaration
int add(int a, int b) {
    return a + b;
}

// Function prototype
void printMessage(int code);

// Main block (no parentheses needed)
main {
    int x = 42;
    float y = 3.14;
    
    if (x > 10 && y <= 5.0) {
        print("Hello World!");
        x++;
    }
    
    // For loop - no declarations allowed in init
    int i = 0;
    for (i = 0; i < 10; i++) {
        print(i);
    }
    
    // While loop
    while (x > 0) {
        x--;
    }
    
    return 0;
}

// Function implementation
void printMessage(int code) {
    if (code == 0) {
        print("Success");
    } else {
        print("Error");
    }
}
