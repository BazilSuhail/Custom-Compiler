const int globalConst = 10;
float globalVar = 2.5;

int add(int a, int b) {
    int result = a + b;
    return result;
}

void loopExample() {
    int i = 0;
    for (i = 0; i < 5; i++) {
        if (i % 2 == 0) {
            continue;
        } else {
            break;
        }
    }
}

void switchExample(int value) {
    switch (value) {
        case 1:
            globalVar = globalVar + 1.0;
            break;
        case 2:
            globalVar = globalVar - 1.0;
        case 3:
            globalVar = globalVar * 2.0;
            break;
        default:
            globalVar = 0.0;
    }
}

int main() {
    int x = 5;
    int y = 10;
    int z = add(x, y);

    while (z > 0) {
        z--;
    }

    do {
        z++;
    } while (z < 5);

    switchExample(z);

    return 0;
}
